44  	/**
45  	 * Checks if graph exists in triple store. Works only when the graph is not
46  	 * empty.
47  	 * 
48  	 * @param graphName
49  	 *            the graph name
50  	 * @return boolean value if specified graph exists and contains at least one
51  	 *         triple elsewhere it will return false
52  	 */
53  	public static boolean checkGraphExistence(final String graphName){
54  		String query = "ASK { GRAPH <" + graphName + "> {?s ?p ?o} }";
55  		return TripleStoreInterfaceSingleton.get().executeAskQuery(query);
56  	}

58  	/**
59  	 * Creates the whole revision from the add and delete sets of the
60  	 * predecessors. Saved in graph tempGraphName.
61  	 * 
62  	 * @param graphName
63  	 *            the graph name
64  	 * @param revisionName
65  	 *            revision number or revision name to build content for
66  	 * @param tempGraphName
67  	 *            the graph where the temporary graph is stored
68  	 * @throws InternalErrorException 
69  	 */
70  	public static void generateFullGraphOfRevision(final String graphName, final String revisionName,
71  			final String tempGraphName) throws InternalErrorException {
72  		logger.info("Rebuild whole content of revision " + revisionName + " of graph <" + graphName
73  				+ "> into temporary graph <" + tempGraphName + ">");
74  		RevisionGraph graph = new RevisionGraph(graphName);
75  		Revision revision = graph.getRevision(revisionName);
76  
77  		// Create temporary graph
78  		TripleStoreInterfaceSingleton.get().executeUpdateQuery("DROP SILENT GRAPH <" + tempGraphName + ">");
79  		TripleStoreInterfaceSingleton.get().executeUpdateQuery("CREATE GRAPH <" + tempGraphName + ">");
80  
81  		// Create path to revision
82  
83  		LinkedList<Revision> list = PathCalculationSingleton.getInstance().getPathToRevisionWithFullGraph(graph, revision)
84  				.getRevisionPath();
85  
86  		// Copy branch to temporary graph
87  		String number = list.removeLast().getRevisionIdentifier();
88  		TripleStoreInterfaceSingleton.get().executeUpdateQuery(
89  				"COPY GRAPH <" + graph.getReferenceGraph(number) + "> TO GRAPH <"
90  						+ tempGraphName + ">");
91  
92  		while (!list.isEmpty()) {
93  			// add- und delete-sets could be extracted from revision tree information
94  			// hard coded variant is faster
95  			String graph_removed = graphName + "-deleteSet-"+ number;
96  			String graph_added   = graphName + "-addSet-"+ number;
97  			// Add data to temporary graph
98  			if (RevisionManagementOriginal.checkGraphExistence(graph_removed))
99  				TripleStoreInterfaceSingleton.get().executeUpdateQuery("ADD GRAPH <" + graph_removed + "> TO GRAPH <" + tempGraphName + ">");
100 			// Remove data from temporary graph (no opposite of SPARQL ADD available)
101 			if (RevisionManagementOriginal.checkGraphExistence(graph_added))
102 				TripleStoreInterfaceSingleton.get().executeUpdateQuery(  "DELETE { GRAPH <" + tempGraphName+ "> { ?s ?p ?o.} }"
103 														+ "WHERE  { GRAPH <" + graph_added	+ "> { ?s ?p ?o.} }");
104 			Revision first = list.removeLast();
105 			if (first!=null)
106 				number = first.getRevisionIdentifier();
107 		}
108 
109 	}

145 	public static void splitAndExecuteBigQuery(final String graphName, final String dataSetAsNTriples, final String template){
146 		final int MAX_STATEMENTS = 500;
147 		String[] lines = dataSetAsNTriples.split("\n");
148 		int counter = 0;
149 		StringBuilder insert = new StringBuilder();
150 		
151 		for (int i=0; i < lines.length; i++) {
152 			insert.append(lines[i]);
153 			insert.append("\n");
154 			counter++;
155 			if (counter == MAX_STATEMENTS-1) {
156 				TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format(template, graphName, insert));
157 				counter = 0;
158 				insert = new StringBuilder();
159 			}
160 		}
161 
162 		TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format(template, graphName, insert));
163 	}

169 	 * Download complete revision information of R43ples from SPARQL endpoint.
170 	 * Provide only information from specified graph if not null
171 	 * 
172 	 * @param graphName
173 	 *            provide only information from specified graph (if not NULL)
174 	 * @param format
175 	 *            serialization of the RDF model
176 	 * @return String containing the RDF model in the specified serialization
177 	 */
178 	public static String getRevisionInformation(final String graphName, final String format) {
179 		String sparqlQuery;
180 		
181 		if (graphName.equals("")) {
182 			sparqlQuery = Config.prefixes + String.format(""
183 					+ "CONSTRUCT { ?s ?p ?o. }"
184 					+ "WHERE { "
185 					+ "	GRAPH <%s> { ?graph a rmo:Graph; rmo:hasRevisionGraph ?revisiongraph.}"
186 					+ "	GRAPH ?revisionGraph {?s ?p ?o.}"
187 					+ "}", Config.revision_graph);
188 			return TripleStoreInterfaceSingleton.get().executeConstructQuery(sparqlQuery, format);
189 		} else {
190 			RevisionGraph graph = new RevisionGraph(graphName);
191 			return graph.getContentOfRevisionGraph(format);
192 		}
193 		
194 	}

207 		return TripleStoreInterfaceSingleton.get().executeConstructQuery(query, format);

261 	/**
262 	 * Get the user URI. If the user does not exist then the user will be created.
263 	 *
264 	 * @param user name as string
265 	 * @return URI of user
266 	 */
267 	public static String getUserURI(final String user) {
268 		// When user does not already exists - create new
269 
270 		String query = Config.prefixes
271 				+ String.format("SELECT ?personUri { GRAPH <%s>  { " + "?personUri a prov:Person;"
272 						+ "  rdfs:label \"%s\"." + "} }", Config.revision_graph, user); //TODO check if the users are created within the special revision graph - maybe move to RevisionGraph
273 		ResultSet results = TripleStoreInterfaceSingleton.get().executeSelectQuery(query);
274 		if (results.hasNext()) {
275 			logger.debug("User " + user + " already exists.");
276 			QuerySolution qs = results.next();
277 			return qs.getResource("?personUri").toString();
278 		} else {
279 			String personUri = null;
280 			try {
281 				personUri = "http://eatld.et.tu-dresden.de/persons/" + URLEncoder.encode(user, "UTF-8");
282 			} catch (UnsupportedEncodingException e) {
283 				e.printStackTrace();
284 			}
285 			logger.debug("User does not exists. Create user " + personUri + ".");
286 			query = Config.prefixes
287 					+ String.format("INSERT DATA { GRAPH <%s> { <%s> a prov:Person; rdfs:label \"%s\". } }",
288 							Config.revision_graph, personUri, user);
289 			TripleStoreInterfaceSingleton.get().executeUpdateQuery(query);
290 			return personUri;
291 		}
292 	}

300 	 * Get the ADD set URI of a given revision URI.
301 	 * 
302 	 * @param revisionURI the revision URI
303 	 * @param revisionGraph the revision graph
304 	 * @return the ADD set URI, returns null when the revision URI does not exists or no ADD set is referenced by the revision URI
305 	 */
306 	public static String getAddSetURI(String revisionURI, String revisionGraph) {
307 		String query = String.format(
308 			  "SELECT ?addSetURI %n"
309 			+ "WHERE { GRAPH <%s> {%n"
310 			+ "	<%s> <http://eatld.et.tu-dresden.de/rmo#addSet> ?addSetURI . %n"
311 			+ "} }", revisionGraph, revisionURI);
312 		
313 		ResultSet results = TripleStoreInterfaceSingleton.get().executeSelectQuery(query);
314 		
315 		if (results.hasNext()) {
316 			QuerySolution qs = results.next();
317 			return qs.getResource("?addSetURI").toString();
318 		} else {
319 			return null;
320 		}
321 	}
