61  	/**
62  	* Read the configuration information from local file.
63  	*
64  	* @param configFilePath path to configuration file
65  	* @throws ConfigurationException
66  	*/
67  	public static void readConfig(final String configFilePath) throws ConfigurationException{
68  		PropertiesConfiguration config;
69  		try {
70  			config = new PropertiesConfiguration(configFilePath);
71  			
72  			triplestore_type = config.getString("triplestore.type");
73  			triplestore_url = config.getString("triplestore.url");
74  			triplestore_user = config.getString("triplestore.user");
75  			triplestore_password = config.getString("triplestore.password");
76  			
77  			service_host = config.getString("service.host");
78  			service_port = config.getInt("service.port");
79  			service_path = config.getString("service.path");
80  			
81  			ssl_keystore = config.getString("ssl.keystore");
82  			ssl_password = config.getString("ssl.password");
83  			
84  			revision_graph = config.getString("revision.graph");
85  			
86  			sdd_graph = config.getString("sdd.graph");
87  			sdd_graph_defaultContent = config.getString("sdd.graph.defaultContent");
88  			
89  			Iterator<String> it = config.getKeys("prefix");
90  			while ( it.hasNext()) {
91  				String prefix = it.next();
92  				String namespace = config.getString(prefix);
93  				prefix = prefix.replace("prefix.", "");
94  				user_defined_prefixes.put(prefix, namespace);
95  			}
96  
97  		} catch (ConfigurationException e) {
98  			logger.warn("Could not read configuration file '" + configFilePath + "'. Switch to 'r43ples.dist.conf'.");
99  			readConfig("r43ples.dist.conf");
100 		}
101 	}

103 	public static String getUserDefinedSparqlPrefixes(){
104 		StringBuilder sb = new StringBuilder();
105 		Set<String> set = user_defined_prefixes.keySet();
106 		Iterator<String> it = set.iterator();
107 		while (it.hasNext()){
108 			String prefix = it.next();
109 			String namespace = user_defined_prefixes.get(prefix);
110 			sb.append("PREFIX "+prefix+": <"+namespace+"> \n");
111 		}
112 		return sb.toString();
113 	}
