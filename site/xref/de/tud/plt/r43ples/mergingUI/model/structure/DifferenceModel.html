3   import java.util.ArrayList;
4   import java.util.HashMap;
5   
6   /**
7    * The difference model provides the structure for storing the returned difference model after a MERGE query.
8    * 
9    * @author Stephan Hensel
10   *
11   */
12  public class DifferenceModel {
13  
14  	/** The hash map which contains all difference groups. **/
15  	private HashMap<String, DifferenceGroup> differenceGroups = new HashMap<String, DifferenceGroup>();
16  	
17  	public ArrayList<Difference> allDifferences = new ArrayList<Difference>(); 
18  	
19  	
20  	/**
21  	 * Add a difference group. If the difference group identifier already exists the old difference group will be overwritten.
22  	 * 
23  	 * @param identifier the identifier
24  	 * @param differenceGroup the difference group
25  	 */
26  	public void addDifferenceGroup(String identifier, DifferenceGroup differenceGroup) {
27  		this.differenceGroups.put(identifier, differenceGroup);
28  	}
29  	
30  	
31  	/**
32  	 * Add a difference group. If the difference group identifier already exists the old difference group will be overwritten.
33  	 * 
34  	 * @param identifier the identifier
35  	 * @param differenceGroup the difference group
36  	 */
37  	public void addDifference(Difference difference) {
38  		this.allDifferences.add(difference);
39  	}
40  	
41  	
42  	/**
43  	 * Remove entry from difference groups.
44  	 * 
45  	 * @param identifier the identifier of the difference group to remove
46  	 */
47  	public void removeDifferenceGroup(String identifier) {
48  		this.differenceGroups.remove(identifier);
49  	}
50  
51  
52  	public HashMap<String, DifferenceGroup> getDifferenceGroups() {
53  		return differenceGroups;
54  	}
55  	
56  	
57  }
