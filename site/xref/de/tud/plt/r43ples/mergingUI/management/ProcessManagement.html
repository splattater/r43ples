36  	/**
37  	 * Read difference model to java representation.
38  	 * 
39  	 * @param differenceModelToRead the difference model to read in turtle serialisation
40  	 * @param differenceModel the difference model where the result should be stored
41  	 * @return 
42  	 */
43  	public static DifferenceModel readDifferenceModel(Model model) {
44  		DifferenceModel differenceModel =  new DifferenceModel();
45  		
46  		
47  		// Query all difference groups
48  		String queryDifferenceGroups = Config.prefixes + 
49  				  "SELECT ?uri ?tripleStateA ?tripleStateB ?automaticResolutionState ?conflicting "
50  				+ "WHERE { "
51  				+ "	?uri a rpo:DifferenceGroup ; "
52  				+ "		sddo:hasTripleStateA ?tripleStateA ; "
53  				+ "		sddo:hasTripleStateB ?tripleStateB ; "
54  				+ "		sddo:automaticResolutionState ?automaticResolutionState ; "
55  				+ "		sddo:isConflicting ?conflicting . "
56  				+ "}";
57  		QueryExecution qeDifferenceGroups = QueryExecutionFactory.create(queryDifferenceGroups, model);
58  		ResultSet resultSetDifferenceGroups = qeDifferenceGroups.execSelect();
59  	    while(resultSetDifferenceGroups.hasNext()) {
60  	    	QuerySolution qsDifferenceGroups = resultSetDifferenceGroups.next();
61  	    	String uri = qsDifferenceGroups.getResource("?uri").toString();
62  	    	SDDTripleStateEnum tripleStateA = convertSDDStringToSDDTripleState(qsDifferenceGroups.getResource("?tripleStateA").toString());
63  	    	SDDTripleStateEnum tripleStateB = convertSDDStringToSDDTripleState(qsDifferenceGroups.getResource("?tripleStateB").toString());
64  	    	SDDTripleStateEnum automaticResolutionState = convertSDDStringToSDDTripleState(qsDifferenceGroups.getResource("?automaticResolutionState").toString());
65  	    	boolean conflicting = qsDifferenceGroups.getLiteral("?conflicting").toString().equals("true^^http://www.w3.org/2001/XMLSchema#boolean");   
66  	    	ResolutionStateEnum resolutionState = ResolutionStateEnum.DIFFERENCE;
67  	    	if (conflicting) {
68  	    		resolutionState = ResolutionStateEnum.CONFLICT;
69  	    	}
70  	    	
71  	    	DifferenceGroup differenceGroup = new DifferenceGroup(tripleStateA, tripleStateB, automaticResolutionState, conflicting);
72  	    	
73  	    	// Query all differences
74  			String queryDifferences = Config.prefixes + String.format(
75  					  "SELECT ?subject ?predicate ?object ?referencedRevisionA ?referencedRevisionB %n"
76  					+ "WHERE { %n"
77  					+ "	<%s> a rpo:DifferenceGroup ; %n"
78  					+ "		rpo:hasDifference ?differenceUri . %n"
79  					+ "	?differenceUri a rpo:Difference ; %n"
80  					+ "		rpo:hasTriple ?tripleUri . %n"
81  					+ "	OPTIONAL { ?differenceUri rpo:referencesA ?referencedRevisionA . } %n"
82  					+ "	OPTIONAL { ?differenceUri rpo:referencesB ?referencedRevisionB . } %n"
83  					+ "	?tripleUri rdf:subject ?subject ; %n"
84  					+ "		rdf:predicate ?predicate ; %n"
85  					+ "		rdf:object ?object . %n"
86  					+ "}", uri);
87  			QueryExecution qeDifferences = QueryExecutionFactory.create(queryDifferences, model);
88  			ResultSet resultSetDifferences = qeDifferences.execSelect();
89  		    while(resultSetDifferences.hasNext()) {
90  		    	QuerySolution qsDifferences = resultSetDifferences.next();
91  		    	
92  		    	String subject = model.qnameFor(qsDifferences.getResource("?subject").toString());
93  		    	String predicate = model.qnameFor(qsDifferences.getResource("?predicate").toString());
94  	    	
95  		    	// Differ between literal and resource
96  				String object = "";
97  				TripleObjectTypeEnum objectType = null;
98  				if (qsDifferences.get("?object").isLiteral()) {
99  					object = qsDifferences.getLiteral("?object").toString();
100 					objectType = TripleObjectTypeEnum.LITERAL;
101 				} else {
102 					object = model.qnameFor(qsDifferences.getResource("?object").toString());
103 					objectType = TripleObjectTypeEnum.RESOURCE;
104 				}
105 		    	
106 		    	Triple triple = new Triple(subject, predicate, object, objectType);
107 		    	
108 		    	String referencedRevisionA = null;
109 		    	if (qsDifferences.getResource("?referencedRevisionA") != null) {
110 		    		referencedRevisionA = qsDifferences.getResource("?referencedRevisionA").toString();
111 		    	}
112 		    	String referencedRevisionB = null;
113 		    	if (qsDifferences.getResource("?referencedRevisionB") != null) {
114 		    		referencedRevisionB = qsDifferences.getResource("?referencedRevisionB").toString();
115 		    	}
116 		    	
117 		    	// Add further information to difference
118 		    	// Get the revision number if available
119 		    	String referencedRevisionLabelA = null;
120 		    	String referencedRevisionLabelB = null;
121 		    	
122 				if ((referencedRevisionA != null) && (referencedRevisionB == null)) {
123 					String query = Config.prefixes + String.format(
124 							  "SELECT ?rev %n"
125 							+ "FROM <%s> %n"
126 							+ "WHERE { %n"
127 							+ "	<%s> a rmo:Revision ; %n"
128 							+ "		rmo:revisionNumber ?rev . %n"
129 							+ "}", Config.revision_graph, referencedRevisionA);
130 					
131 					ResultSet resultSet = TripleStoreInterfaceSingleton.get().executeSelectQuery(query);
132 					
133 					if (resultSet.hasNext()) {
134 						QuerySolution qs = resultSet.next();
135 						referencedRevisionLabelA = qs.getLiteral("?rev").toString();
136 					}
137 				} else if ((referencedRevisionA == null) && (referencedRevisionB != null)) {
138 					String query = Config.prefixes + String.format(
139 							  "SELECT ?rev %n"
140 							+ "FROM <%s> %n"
141 							+ "WHERE { %n"
142 							+ "	<%s> a rmo:Revision ; %n"
143 							+ "		rmo:revisionNumber ?rev . %n"
144 							+ "}", Config.revision_graph, referencedRevisionB);
145 					
146 					ResultSet resultSet = TripleStoreInterfaceSingleton.get().executeSelectQuery(query);
147 					
148 					// Iterate over all labels
149 					if (resultSet.hasNext()) {
150 						QuerySolution qs = resultSet.next();
151 						referencedRevisionLabelB = qs.getLiteral("?rev").toString();
152 					}
153 				} else if ((referencedRevisionA != null) && (referencedRevisionB != null)) {
154 					String query = Config.prefixes + String.format(
155 							  "SELECT ?revA ?revB %n"
156 							+ "FROM <%s> %n"
157 							+ "WHERE { %n"
158 							+ "	<%s> a rmo:Revision ; %n"
159 							+ "		rmo:revisionNumber ?revA . %n"
160 							+ "	<%s> a rmo:Revision ; %n"
161 							+ "		rmo:revisionNumber ?revB . %n"
162 							+ "}", Config.revision_graph, referencedRevisionA, referencedRevisionB);
163 					
164 					ResultSet resultSet = TripleStoreInterfaceSingleton.get().executeSelectQuery(query);
165 		
166 					// Iterate over all labels
167 					if (resultSet.hasNext()) {
168 						QuerySolution qs = resultSet.next();
169 						referencedRevisionLabelA = qs.getLiteral("?revA").toString();
170 						referencedRevisionLabelB = qs.getLiteral("?revB").toString();
171 					}
172 				}	    	
173 		    	
174 		    	Difference difference = new Difference(triple, 
175 		    			referencedRevisionA, referencedRevisionLabelA, tripleStateA, 
176 		    			referencedRevisionB, referencedRevisionLabelB, tripleStateB,
177 		    			conflicting, automaticResolutionState, resolutionState);
178 		    	differenceGroup.addDifference(triple, difference);
179 		    	differenceModel.addDifference(difference);
180 		    }
181 	    	differenceModel.addDifferenceGroup(differenceGroup.getTripleStateA().toString() + "-" + differenceGroup.getTripleStateB().toString(), differenceGroup);
182 	    }
183 	    
184 	    logger.info("Difference model successfully read.");	
185 		return differenceModel;
186 	}

191 	/**
192 	 * Convert SDD state string to SDD triple state. If value does not exists in enum null will be returned.
193 	 * 
194 	 * @param state the state to convert
195 	 * @return the SDD triple state
196 	 */
197 	
198 	private static SDDTripleStateEnum convertSDDStringToSDDTripleState(String state) {
199 		if (state.equals(SDDTripleStateEnum.ADDED.getSddRepresentation())) {
200 			return SDDTripleStateEnum.ADDED;
201 		} else if (state.equals(SDDTripleStateEnum.DELETED.getSddRepresentation())) {
202 			return SDDTripleStateEnum.DELETED;
203 		} else if (state.equals(SDDTripleStateEnum.ORIGINAL.getSddRepresentation())) {
204 			return SDDTripleStateEnum.ORIGINAL;
205 		} else if (state.equals(SDDTripleStateEnum.NOTINCLUDED.getSddRepresentation())) {
206 			return SDDTripleStateEnum.NOTINCLUDED;
207 		} else {
208 			return null;
209 		}
210 	}

214 	/**
215 	 * Get difference by triple. If the difference model does not contain the triple null will be returned.
216 	 * 
217 	 * @param triple the triple to look for
218 	 * @param differenceModel the difference model
219 	 * @return the difference or null if triple is not included in difference model
220 	 */
221 	public static boolean isConflicting(Triple triple, DifferenceModel differenceModel) {
222 		if (differenceModel.allDifferences.contains(triple)){
223 			int pos = differenceModel.allDifferences.indexOf(triple);
224 			return differenceModel.allDifferences.get(pos).conflicting;
225 		}
226 		else 
227 			return false;
228 	}

232 	/**create High level change Table Model from high level change Model
233 	 * @param highLevelChangeModel the model to be read 
234 	 * @return hightLevelChangeTableModel the Model to be created 
235 	 *  */
236 	public static HighLevelChangeTableModel createHighLevelChangeTableModel (HighLevelChangeModel highLevelChangeModel) {
237 		HighLevelChangeTableModel highLevelChangeTableModel = new HighLevelChangeTableModel();
238 		Iterator<Entry<String, HighLevelChangeRenaming>> iterHL = highLevelChangeModel.getHighLevelChangesRenaming().entrySet().iterator();
239 		while(iterHL.hasNext()){
240 			HighLevelChangeRenaming hlcr = iterHL.next().getValue();
241 			//get added Difference and deleted Difference
242 			Difference additionDifference = hlcr.getAdditionDifference();
243 			Difference deletionDifference = hlcr.getDeletionDifference();
244 			Triple additionTriple = additionDifference.getTriple();
245 			Triple deletionTriple = deletionDifference.getTriple();
246 					
247 			String subject = additionTriple.getSubject();
248 			String predicate = additionTriple.getPredicate();
249 			String altObject = deletionTriple.getObject();
250 			String newObject = additionTriple.getObject();
251 			
252 			//get checkBox state
253 			//get approved button state
254 			String isRenaming = "no";
255 			String isResolved = "no";
256 			if(additionDifference.getResolutionState() == ResolutionStateEnum.RESOLVED
257 					&& deletionDifference.getResolutionState() == ResolutionStateEnum.RESOLVED) {
258 				if(additionDifference.getTripleResolutionState() == SDDTripleStateEnum.ADDED
259 						&& deletionDifference.getTripleResolutionState() == SDDTripleStateEnum.DELETED) {
260 					isRenaming = "yes";
261 					isResolved = "yes";	
262 				} else if (additionDifference.getTripleResolutionState() == SDDTripleStateEnum.DELETED 
263 						&& deletionDifference.getTripleResolutionState() == SDDTripleStateEnum.ADDED) {
264 					isRenaming = "no";
265 					isResolved = "yes";
266 				}
267 			}else if(additionDifference.getTripleResolutionState() == SDDTripleStateEnum.ADDED
268 					&& deletionDifference.getTripleResolutionState() == SDDTripleStateEnum.DELETED ){
269 				isRenaming = "yes";
270 			}
271 			
272 			highLevelChangeTableModel.readTableRow(new HighLevelChangeTableRow(hlcr, subject, predicate, altObject, newObject, isResolved, isRenaming));
273 		}
274 		return highLevelChangeTableModel;
275 	}
