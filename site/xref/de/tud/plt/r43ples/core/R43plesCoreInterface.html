1   package de.tud.plt.r43ples.core;
2   
3   import de.tud.plt.r43ples.exception.InternalErrorException;
4   import de.tud.plt.r43ples.exception.QueryErrorException;
5   import de.tud.plt.r43ples.existentobjects.*;
6   import de.tud.plt.r43ples.management.R43plesRequest;
7   
8   import java.util.ArrayList;
9   
10  /**
11   * This interface provides methods to access the core functions of R43ples.
12   *
13   * @author Stephan Hensel
14   *
15   */
16  public interface R43plesCoreInterface {
17  
18      /**
19       * Create a new initial commit.
20       *
21       * @param request the request received by R43ples
22       * @return the created initial commit
23       * @throws InternalErrorException
24       */
25      InitialCommit createInitialCommit(R43plesRequest request) throws InternalErrorException;
26  
27      /**
28       * Create a new initial commit.
29       *
30       * @param graphName the graph name
31       * @param addSet the add set as N-Triples
32       * @param deleteSet the delete set as N-Triples
33       * @param user the user
34       * @param message the message
35       * @return the created update commit
36       * @throws InternalErrorException
37       */
38      InitialCommit createInitialCommit(String graphName, String addSet, String deleteSet, String user, String message) throws InternalErrorException;
39  
40  
41      /**
42       * Create a new update commit.
43       *
44       * @param request the request received by R43ples
45       * @return the list of created update commits
46       * @throws InternalErrorException
47       */
48      ArrayList<UpdateCommit> createUpdateCommit(R43plesRequest request) throws InternalErrorException;
49  
50      /**
51       * Create a new update commit.
52       *
53       * @param graphName the graph name
54       * @param addSet the add set as N-Triples
55       * @param deleteSet the delete set as N-Triples
56       * @param user the user
57       * @param message the message
58       * @param derivedFromIdentifier the revision identifier of the revision or the reference identifier from which the new revision should be derive from
59       * @return the created update commit
60       * @throws InternalErrorException
61       */
62      UpdateCommit createUpdateCommit(String graphName, String addSet, String deleteSet, String user, String message, String derivedFromIdentifier) throws InternalErrorException;
63  
64      /**
65       * Create a new reference commit.
66       *
67       * @param request the request received by R43ples
68       * @return the created reference commit
69       * @throws InternalErrorException
70       */
71      ReferenceCommit createReferenceCommit(R43plesRequest request) throws InternalErrorException;
72  
73      /**
74       * Create a new reference commit.
75       *
76       * @param graphName the graph name
77       * @param referenceName the reference name
78       * @param revisionIdentifier the revision identifier (the corresponding revision will be the current base for the reference)
79       * @param user the user
80       * @param message the message
81       * @param isBranch states if the created reference is a branch or a tag. (branch => true; tag => false)
82       * @return the created reference commit
83       * @throws InternalErrorException
84       */
85      ReferenceCommit createReferenceCommit(String graphName, String referenceName, String revisionIdentifier, String user, String message, boolean isBranch) throws InternalErrorException;
86  
87      /**
88       * Create a new merge commit.
89       *
90       * @param request the request received by R43ples
91       * @return the created merge commit
92       * @throws InternalErrorException
93       */
94      MergeCommit createMergeCommit(R43plesRequest request) throws InternalErrorException;
95  
96      /**
97       * Creates a three way merge commit draft by using the corresponding meta information.
98       *
99       * @param graphName the graph name
100      * @param branchNameFrom the branch name (from)
101      * @param branchNameInto the branch name (into)
102      * @param user the user
103      * @param message the message
104      * @param sdd the SDD URI to use
105      * @param triples the triples of the query WITH part
106      * @param type the query type (FORCE, AUTO, MANUAL)
107      * @param with states if the WITH part is available
108      * @throws InternalErrorException
109      */
110     ThreeWayMergeCommit createThreeWayMergeCommit(String graphName, String branchNameFrom, String branchNameInto, String user, String message, String sdd, String triples, MergeTypes type, boolean with) throws InternalErrorException;
111 
112     /**
113      * Create a new pick commit.
114      *
115      * @param request the request received by R43ples
116      * @return the created pick commit
117      * @throws InternalErrorException
118      */
119     PickCommit createPickCommit(R43plesRequest request) throws InternalErrorException;
120 
121     /**
122      * Drop graph query. This query will delete the whole graph and all corresponding revision information.
123      *
124      * @param query the query
125      * @throws QueryErrorException
126      */
127     void sparqlDropGraph(final String query) throws QueryErrorException;
128 
129     /**
130      * Get the response of a SPARQL query (SELECT, CONSTRUCT, ASK).
131      *
132      * @param request the request
133      * @param query_rewriting option if query rewriting should be enabled (true => enabled)
134      * @return the query response
135      * @throws InternalErrorException
136      */
137     String getSparqlSelectConstructAskResponse(final R43plesRequest request, final boolean query_rewriting) throws InternalErrorException;
138 }
