48      /**
49       * Tries to create a new commit draft as a new commit in the triple store.
50       * If possible it will create the corresponding revision and the meta data.
51       *
52       * @return the commit (has attribute which indicates if the commit was executed or not)
53       */
54      protected ThreeWayMergeCommit createCommitInTripleStore() throws InternalErrorException {
55  
56          String revisionGraphURI = getRevisionGraph().getRevisionGraphUri();
57          String revisionUriFrom = getRevisionGraph().getRevisionUri(getBranchNameFrom());
58          String revisionUriInto = getRevisionGraph().getRevisionUri(getBranchNameInto());
59  
60          // Differ between MERGE query with specified SDD and without SDD
61          String usedSDDURI = getRevisionGraph().getSDD(getSdd());
62  
63          // Get the common revision with shortest path
64          Revision commonRevision = this.getPathCalculationInterface().getCommonRevisionWithShortestPath(getRevisionGraph(), new Revision(getRevisionGraph(), revisionUriFrom, false), new Revision(getRevisionGraph(), revisionUriInto, false));
65  
66          // Create the revision progress for from and into
67          String namedGraphUriFrom = getRevisionManagement().getTemporaryRevisionProgressFromURI(getRevisionGraph());
68          String namedGraphUriInto = getRevisionManagement().getTemporaryRevisionProgressIntoURI(getRevisionGraph());
69          String namedGraphUriDiff = getRevisionManagement().getTemporaryDifferenceModelURI(getRevisionGraph());
70          String uriA = "http://eatld.et.tu-dresden.de/branch-from";
71          String uriB = "http://eatld.et.tu-dresden.de/branch-into";
72  
73          Revision fromRevision = new Revision(getRevisionGraph(), revisionUriFrom, false);
74          Revision intoRevision = new Revision(getRevisionGraph(), revisionUriInto, false);
75  
76          createRevisionProgresses(revisionGraphURI, getGraphName(),
77                  this.getPathCalculationInterface().getPathBetweenStartAndTargetRevision(getRevisionGraph(), commonRevision, fromRevision),
78                  namedGraphUriFrom, uriA,
79                  this.getPathCalculationInterface().getPathBetweenStartAndTargetRevision(getRevisionGraph(), commonRevision, intoRevision),
80                  namedGraphUriInto, uriB, commonRevision);
81  
82          // Create difference model
83          createDifferenceTripleModel(getGraphName(), namedGraphUriDiff, namedGraphUriFrom, uriA, namedGraphUriInto, uriB,
84                  usedSDDURI);
85  
86          // The created revision
87          Revision revision;
88  
89          // Differ between the different merge queries
90          if ((getType() != null) && (getType().equals(MergeTypes.AUTO)) && !isWith()) {
91              logger.debug("AUTO MERGE query detected");
92              // Create the merged revision
93              revision = createMergedRevision(namedGraphUriDiff, MergeQueryTypeEnum.AUTO);
94              return addMetaInformation(revision, namedGraphUriDiff, commonRevision, fromRevision, intoRevision);
95          } else if ((getType() != null) && (getType().equals(MergeTypes.MANUAL)) && isWith()) {
96              logger.debug("MANUAL MERGE query detected");
97              // Create the merged revision
98              revision = createMergedRevision(namedGraphUriDiff, MergeQueryTypeEnum.MANUAL);
99              return addMetaInformation(revision, namedGraphUriDiff, commonRevision, fromRevision, intoRevision);
100         } else if ((getType() == null) && isWith()) {
101             logger.debug("MERGE WITH query detected");
102             // Create the merged revision
103             revision = createMergedRevision(namedGraphUriDiff, MergeQueryTypeEnum.WITH);
104             return addMetaInformation(revision, namedGraphUriDiff, commonRevision, fromRevision, intoRevision);
105         } else if ((getType() == null) && !isWith()) {
106             logger.debug("MERGE query detected");
107             // Check if difference model contains conflicts
108             String queryASK = String.format("ASK { %n" + "	GRAPH <%s> { %n"
109                     + " 	?ref <http://eatld.et.tu-dresden.de/sddo#isConflicting> \"true\"^^<http://www.w3.org/2001/XMLSchema#boolean> . %n"
110                     + "	} %n" + "}", namedGraphUriDiff);
111             if (getTripleStoreInterface().executeAskQuery(queryASK)) {
112                 // Difference model contains conflicts
113                 // Return the conflict model to the client
114                 String conflictModel = RevisionManagementOriginal.getContentOfGraph(namedGraphUriDiff, "text/turtle");
115                 return new ThreeWayMergeCommit(getRevisionGraph(), null,null, null, null, fromRevision, null, intoRevision, null, null, null, true, conflictModel, namedGraphUriDiff);
116             } else {
117                 // Difference model contains no conflicts
118                 // Create the merged revision
119                 revision = createMergedRevision(namedGraphUriDiff, MergeQueryTypeEnum.COMMON);
120                 return addMetaInformation(revision, namedGraphUriDiff, commonRevision, fromRevision, intoRevision);
121             }
122         } else {
123             throw new InternalErrorException("This is not a valid MERGE query");
124         }
125     }

127     /**
128      * Adds meta information for commit and revision to the revision graph.
129      *
130      * <img src="{@docRoot}../../doc/revision management description/r43ples-threewaymerge.png" />
131      *
132      * @param generatedRevision the generated revision
133      * @param namedGraphUriDiff the named graph URI of the difference model
134      * @param commonRevision the common revision of this merge
135      * @param usedSourceRevision the used source revision (from)
136      * @param usedTargetRevision the used target revision (into)
137      * @return the created commit
138      * @throws InternalErrorException
139      */
140     private ThreeWayMergeCommit addMetaInformation(Revision generatedRevision, String namedGraphUriDiff, Revision commonRevision, Revision usedSourceRevision, Revision usedTargetRevision) throws InternalErrorException {
141 
142         String commitURI = getRevisionManagement().getNewThreeWayMergeCommitURI(getRevisionGraph(), generatedRevision.getRevisionIdentifier());;
143 
144         Branch usedSourceBranch = getRevisionGraph().getBranch(getBranchNameFrom(), true);
145         Branch usedTargetBranch = getRevisionGraph().getBranch(getBranchNameInto(), true);
146 
147         String personUri = RevisionManagementOriginal.getUserURI(getUser());
148 
149         // Create a new commit (activity)
150         StringBuilder queryContent = new StringBuilder(1000);
151         queryContent.append(String.format(
152                 "<%s> a rmo:ThreeWayMergeCommit, rmo:MergeCommit, rmo:Commit; "
153                         + "	prov:wasAssociatedWith <%s> ;"
154                         + "	dc-terms:title \"%s\" ;"
155                         + "	prov:atTime \"%s\"^^xsd:dateTime ; %n"
156                         + " prov:generated <%s> ;"
157                         + " rmo:usedSourceRevision <%s> ;"
158                         + " rmo:usedSourceBranch <%s> ;"
159                         + " rmo:usedTargetRevision <%s> ;"
160                         + " rmo:usedTargetBranch <%s> .",
161                 commitURI, personUri, getMessage(), getTimeStamp(),
162                 generatedRevision.getRevisionURI(), usedSourceRevision.getRevisionURI(), usedSourceBranch.getReferenceURI(),
163                 usedTargetRevision.getRevisionURI(), usedTargetBranch.getReferenceURI()));
164 
165         // Create revision meta data
166         queryContent.append(String.format(
167                 "<%s> a rmo:Revision ; %n"
168                         + "	rmo:addSet <%s> ; %n"
169                         + "	rmo:deleteSet <%s> ; %n"
170                         + "	rmo:revisionNumber \"%s\" ; %n"
171                         + "	rmo:belongsTo <%s> ; %n"
172                         + " prov:wasDerivedFrom <%s>, <%s> . %n",
173                 generatedRevision.getRevisionURI(), generatedRevision.getAddSetURI(), generatedRevision.getDeleteSetURI(), generatedRevision.getRevisionIdentifier(),
174                 usedTargetBranch.getReferenceURI(), usedSourceRevision.getRevisionURI(), usedTargetRevision.getRevisionURI()));
175 
176         String query = Config.prefixes
177                 + String.format("INSERT DATA { GRAPH <%s> { %s } }", getRevisionGraph().getRevisionGraphUri(),
178                 queryContent.toString());
179 
180         getTripleStoreInterface().executeUpdateQuery(query);
181 
182         // Move branch to new revision
183         moveBranchReference(getRevisionGraph().getRevisionGraphUri(), usedTargetBranch.getReferenceURI(), usedTargetRevision.getRevisionURI(), generatedRevision.getRevisionURI());
184         // Update the target branch object
185         usedTargetBranch = getRevisionGraph().getBranch(getBranchNameInto(), true);
186 
187         return new ThreeWayMergeCommit(getRevisionGraph(), commitURI, getUser(), getTimeStamp(), getMessage(),
188                 usedSourceRevision, usedSourceBranch, usedTargetRevision, usedTargetBranch, generatedRevision,
189                 commonRevision, false, null, namedGraphUriDiff);
190     }

192     /**
193      * Create a merged revision with meta data
194      *
195      * @param graphNameDifferenceTripleModel the graph name of the difference triple model
196      * @param type the merge query type
197      * @return the created revision
198      * @throws InternalErrorException
199      */
200     private Revision createMergedRevision(String graphNameDifferenceTripleModel, MergeQueryTypeEnum type) throws InternalErrorException {
201 
202         // Create an empty temporary graph which will contain the merged full content
203         String graphNameOfMerged = getRevisionManagement().getTemporaryMergedURI(getRevisionGraph());
204         getTripleStoreInterface().executeUpdateQuery(String.format("DROP SILENT GRAPH <%s>", graphNameOfMerged));
205         getTripleStoreInterface().executeCreateGraph(graphNameOfMerged);
206 
207         // Get the full graph name of branch A
208         String graphNameOfBranchA = getRevisionGraph().getReferenceGraph(getBranchNameFrom());
209         // Get the full graph name of branch B
210         String graphNameOfBranchB = getRevisionGraph().getReferenceGraph(getBranchNameInto());
211 
212         if (type.equals(MergeQueryTypeEnum.MANUAL)) {
213             // Manual merge query
214             RevisionManagementOriginal.executeINSERT(graphNameOfMerged, getTriples());
215         } else {
216             // Copy graph B to temporary merged graph
217             String queryCopy = String.format("COPY <%s> TO <%s>", graphNameOfBranchB, graphNameOfMerged);
218             getTripleStoreInterface().executeUpdateQuery(queryCopy);
219 
220             // Get the triples from branch A which should be added to/removed from the merged revision
221             String triplesToAdd = "";
222             String triplesToDelete = "";
223 
224             // Get all difference groups
225             String queryDifferenceGroup = Config.prefixes + String.format(
226                     "SELECT ?differenceCombinationURI ?automaticResolutionState ?tripleStateA ?tripleStateB ?conflict %n"
227                             + "WHERE { GRAPH <%s> { %n"
228                             + "	?differenceCombinationURI a rpo:DifferenceGroup ; %n"
229                             + "		sddo:automaticResolutionState ?automaticResolutionState ; %n"
230                             + "		sddo:hasTripleStateA ?tripleStateA ; %n"
231                             + "		sddo:hasTripleStateB ?tripleStateB ; %n"
232                             + "		sddo:isConflicting ?conflict . %n"
233                             + "} }", graphNameDifferenceTripleModel);
234 
235             // Iterate over all difference groups
236             ResultSet resultSetDifferenceGroups = getTripleStoreInterface().executeSelectQuery(queryDifferenceGroup);
237             while (resultSetDifferenceGroups.hasNext()) {
238                 QuerySolution qsCurrentDifferenceGroup = resultSetDifferenceGroups.next();
239 
240                 String currentDifferencGroupURI = qsCurrentDifferenceGroup.getResource("?differenceCombinationURI").toString();
241                 String currentDifferencGroupAutomaticResolutionState = qsCurrentDifferenceGroup.getResource("?automaticResolutionState").toString();
242 //				Currently not needed
243 //				String currentDifferencGroupTripleStateA = qsCurrentDifferenceGroup.getResource("?tripleStateA").toString();
244 //				String currentDifferencGroupTripleStateB = qsCurrentDifferenceGroup.getResource("?tripleStateB").toString();
245                 boolean currentDifferencGroupConflict = qsCurrentDifferenceGroup.getLiteral("?conflict").getBoolean();
246 
247                 // Get all differences (triples) of current difference group
248                 String queryDifference = Config.prefixes + String.format(
249                         "SELECT ?s ?p ?o %n"
250                                 + "WHERE { GRAPH <%s> { %n"
251                                 + "	<%s> a rpo:DifferenceGroup ; %n"
252                                 + "		rpo:hasDifference ?blankDifference . %n"
253                                 + "	?blankDifference a rpo:Difference ; %n"
254                                 + "		rpo:hasTriple ?triple . %n"
255                                 + "	?triple rdf:subject ?s . %n"
256                                 + "	?triple rdf:predicate ?p . %n"
257                                 + "	?triple rdf:object ?o . %n"
258                                 + "} }", graphNameDifferenceTripleModel, currentDifferencGroupURI);
259 
260                 // Iterate over all differences (triples)
261                 ResultSet resultSetDifferences = getTripleStoreInterface().executeSelectQuery(queryDifference);
262                 while (resultSetDifferences.hasNext()) {
263                     QuerySolution qsCurrentDifference = resultSetDifferences.next();
264 
265                     String subject = "<" + qsCurrentDifference.getResource("?s").toString() + ">";
266                     String predicate = "<" + qsCurrentDifference.getResource("?p").toString() + ">";
267 
268                     // Differ between literal and resource
269                     String object;
270                     if (qsCurrentDifference.get("?o").isLiteral()) {
271                         object = "\"" + qsCurrentDifference.getLiteral("?o").toString() + "\"";
272                     } else {
273                         object = "<" + qsCurrentDifference.getResource("?o").toString() + ">";
274                     }
275 
276                     if (	type.equals(MergeQueryTypeEnum.AUTO) ||
277                             type.equals(MergeQueryTypeEnum.COMMON) ||
278                             (type.equals(MergeQueryTypeEnum.WITH) && !currentDifferencGroupConflict) ) {
279                         // MERGE AUTO or common MERGE query
280                         if (currentDifferencGroupAutomaticResolutionState.equals(SDDTripleStateEnum.ADDED.getSddRepresentation())) {
281                             // Triple should be added
282                             triplesToAdd += subject + " " + predicate + " " + object + " . \n";
283                         } else {
284                             // Triple should be deleted
285                             triplesToDelete += subject + " " + predicate + " " + object + " . \n";
286                         }
287                     } else {
288                         // MERGE WITH query - conflicting triple
289                         Model model = JenaModelManagement.readNTripleStringToJenaModel(getTriples());
290                         // Create ASK query which will check if the model contains the specified triple
291                         String queryAsk = String.format(
292                                 "ASK { %n"
293                                         + " %s %s %s %n"
294                                         + "}", subject, predicate, object);
295                         Query query = QueryFactory.create(queryAsk);
296                         QueryExecution qe = QueryExecutionFactory.create(query, model);
297                         boolean resultAsk = qe.execAsk();
298                         qe.close();
299                         model.close();
300                         if (resultAsk) {
301                             // Model contains the specified triple
302                             // Triple should be added
303                             triplesToAdd += subject + " " + predicate + " " + object + " . \n";
304                         } else {
305                             // Triple should be deleted
306                             triplesToDelete += subject + " " + predicate + " " + object + " . \n";
307                         }
308                     }
309                 }
310                 // Update the merged graph
311                 // Insert triplesToAdd
312                 RevisionManagementOriginal.executeINSERT(graphNameOfMerged, triplesToAdd);
313                 // Delete triplesToDelete
314                 RevisionManagementOriginal.executeDELETE(graphNameOfMerged, triplesToDelete);
315             }
316         }
317 
318         // Calculate the add and delete sets
319 
320         // Get all added triples (concatenate all triples which are in MERGED but not in A and all triples which are in MERGED but not in B)
321         String queryAddedTriples = String.format(
322                 "CONSTRUCT {?s ?p ?o} %n"
323                         + "WHERE { %n"
324                         + "	GRAPH <%s> { ?s ?p ?o } %n"
325                         + "	FILTER NOT EXISTS { "
326                         + "		GRAPH <%s> { ?s ?p ?o } %n"
327                         + "	} %n"
328                         + "}", graphNameOfMerged, graphNameOfBranchA);
329 
330         String addedTriples = getTripleStoreInterface().executeConstructQuery(queryAddedTriples, FileUtils.langNTriple);
331 
332         queryAddedTriples = String.format(
333                 "CONSTRUCT {?s ?p ?o} %n"
334                         + "WHERE { %n"
335                         + "	GRAPH <%s> { ?s ?p ?o } %n"
336                         + "	FILTER NOT EXISTS { %n"
337                         + "		GRAPH <%s> { ?s ?p ?o } %n"
338                         + "	} %n"
339                         + "}", graphNameOfMerged, graphNameOfBranchB);
340 
341         addedTriples += getTripleStoreInterface().executeConstructQuery(queryAddedTriples, FileUtils.langNTriple);
342 
343         // Get all removed triples (concatenate all triples which are in A but not in MERGED and all triples which are in B but not in MERGED)
344         String queryRemovedTriples = String.format(
345                 "CONSTRUCT {?s ?p ?o} %n"
346                         + "WHERE { %n"
347                         + "	GRAPH <%s> { ?s ?p ?o } %n"
348                         + "	FILTER NOT EXISTS { %n"
349                         + "		GRAPH <%s> { ?s ?p ?o } %n"
350                         + "	} %n"
351                         + "}", graphNameOfBranchA, graphNameOfMerged);
352 
353         String deletedTriples = getTripleStoreInterface().executeConstructQuery(queryRemovedTriples, FileUtils.langNTriple);
354 
355         queryRemovedTriples = String.format(
356                 "CONSTRUCT {?s ?p ?o} %n"
357                         + "WHERE { %n"
358                         + "	GRAPH <%s> { ?s ?p ?o } %n"
359                         + "	FILTER NOT EXISTS { %n"
360                         + "		GRAPH <%s> { ?s ?p ?o } %n"
361                         + "	} %n"
362                         + "}", graphNameOfBranchB, graphNameOfMerged);
363 
364         deletedTriples += getTripleStoreInterface().executeConstructQuery(queryRemovedTriples, FileUtils.langNTriple);
365 
366         // Creates a new revision draft an creates a corresponding revision - no meta data will be written
367         RevisionDraft revisionDraft = new RevisionDraft(getRevisionManagement(), getRevisionGraph(), getBranchNameInto(), addedTriples, deletedTriples);
368 
369         return revisionDraft.createRevisionInTripleStore();
370     }

372     /**
373      * Create the revision progresses for both branches.
374      *
375      * @param pathFrom the path with all revisions from start revision to target revision of the from branch
376      * @param graphNameRevisionProgressFrom the graph name of the revision progress of the from branch
377      * @param uriFrom the URI of the revision progress of the from branch
378      * @param pathInto the linked list with all revisions from start revision to target revision of the into branch
379      * @param graphNameRevisionProgressInto the graph name of the revision progress of the into branch
380      * @param uriInto the URI of the revision progress of the into branch
381      * @throws InternalErrorException
382      */
383     protected void createRevisionProgresses(final String revisionGraph, final String graphName,
384                                             Path pathFrom, String graphNameRevisionProgressFrom, String uriFrom,
385                                             Path pathInto, String graphNameRevisionProgressInto, String uriInto, Revision commonRevision) throws InternalErrorException {
386         logger.info("Create the revision progress of branch from and into.");
387 
388         RevisionGraph graph = new RevisionGraph(graphName);
389 
390         if (!((pathFrom.getRevisionPath().size() > 0) && (pathInto.getRevisionPath().size() > 0))) {
391             throw new InternalErrorException("Revision path contains no revisions.");
392         }
393 
394         // Get the full graph name of common revision or create full revision graph of common revision
395         String fullGraphNameCommonRevision;
396         Boolean tempGraphWasCreated = false;
397         try {
398             fullGraphNameCommonRevision = graph.getReferenceGraph(commonRevision.getRevisionIdentifier());
399         } catch (InternalErrorException e) {
400             // Create a temporary full graph
401             // TODO move to new RevisionManagement
402             fullGraphNameCommonRevision = graphName + "RM-TEMP-REVISION-PROGRESS-FULLGRAPH";
403             RevisionManagementOriginal.generateFullGraphOfRevision(graphName, commonRevision.getRevisionIdentifier(), fullGraphNameCommonRevision);
404             tempGraphWasCreated = true;
405         }
406 
407         // Create revision progress of branch from
408         createRevisionProgress(revisionGraph, pathFrom, fullGraphNameCommonRevision, graphNameRevisionProgressFrom, uriFrom);
409 
410         // Create revision progress of branch into
411         createRevisionProgress(revisionGraph, pathInto, fullGraphNameCommonRevision, graphNameRevisionProgressInto, uriInto);
412 
413         // Drop the temporary full graph
414         if (tempGraphWasCreated) {
415             logger.info("Drop the temporary full graph.");
416             TripleStoreInterfaceSingleton.get().executeUpdateQuery("DROP SILENT GRAPH <" + fullGraphNameCommonRevision + ">");
417         }
418 
419     }

421     /**
422      * Create the revision progress.
423      *
424      * @param path the path with all revisions from start revision to target revision
425      * @param fullGraphNameCommonRevision the full graph name of the common revision (first revision of path)
426      * @param graphNameRevisionProgress the graph name of the revision progress
427      * @param uri the URI of the revision progress
428      * @throws InternalErrorException
429      */
430     private void createRevisionProgress(final String revisionGraph, Path path, String fullGraphNameCommonRevision, String graphNameRevisionProgress, String uri) throws InternalErrorException {
431         logger.info("Create the revision progress of " + uri + " in graph " + graphNameRevisionProgress + ".");
432 
433         TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format("DROP SILENT GRAPH <%s>", graphNameRevisionProgress));
434         TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format("CREATE GRAPH  <%s>", graphNameRevisionProgress));
435         Iterator<Revision> iteList = path.getRevisionPath().iterator();
436 
437         if (iteList.hasNext()) {
438             String firstRevision = iteList.next().getRevisionURI();
439 
440             // Create the initial content
441             logger.info("Create the initial content.");
442             String queryInitial = Config.prefixes + String.format(
443                     "INSERT { GRAPH <%s> { %n"
444                             + "	<%s> a rpo:RevisionProgress; %n"
445                             + "		rpo:original [ %n"
446                             + "			rdf:subject ?s ; %n"
447                             + "			rdf:predicate ?p ; %n"
448                             + "			rdf:object ?o ; %n"
449                             + "			rmo:references <%s> %n"
450                             + "		] %n"
451                             + "} } WHERE { %n"
452                             + "	GRAPH <%s> %n"
453                             + "		{ ?s ?p ?o . } %n"
454                             + "}", graphNameRevisionProgress, uri, firstRevision, fullGraphNameCommonRevision);
455 
456             // Execute the query which generates the initial content
457             TripleStoreInterfaceSingleton.get().executeUpdateQuery(queryInitial);
458 
459             // Update content by current add and delete set - remove old entries
460             while (iteList.hasNext()) {
461                 String revision = iteList.next().getRevisionURI();
462                 logger.info("Update content by current add and delete set of revision " + revision + " - remove old entries.");
463                 // Get the ADD and DELETE set URIs
464                 String addSetURI = RevisionManagementOriginal.getAddSetURI(revision, revisionGraph);
465                 String deleteSetURI = RevisionManagementOriginal.getDeleteSetURI(revision, revisionGraph);
466 
467                 if ((addSetURI != null) && (deleteSetURI != null)) {
468 
469                     // Update the revision progress with the data of the current revision ADD set
470 
471                     // Delete old entries (original)
472                     String queryRevision = Config.prefixes + String.format(
473                             "DELETE { GRAPH <%s> { %n"
474                                     + "	<%s> rpo:original ?blank . %n"
475                                     + "	?blank rdf:subject ?s . %n"
476                                     + "	?blank rdf:predicate ?p . %n"
477                                     + "	?blank rdf:object ?o . %n"
478                                     + "	?blank rmo:references ?revision . %n"
479                                     + "} } %n"
480                                     + "WHERE { "
481                                     + "		GRAPH <%s> { %n"
482                                     + "			<%s> rpo:original ?blank . %n"
483                                     + "			?blank rdf:subject ?s . %n"
484                                     + "			?blank rdf:predicate ?p . %n"
485                                     + "			?blank rdf:object ?o . %n"
486                                     + "			?blank rmo:references ?revision . %n"
487                                     + "		} %n"
488                                     + "		GRAPH <%s> { %n"
489                                     + "			?s ?p ?o %n"
490                                     + "		} %n"
491                                     + "};", graphNameRevisionProgress, uri, graphNameRevisionProgress, uri, addSetURI);
492 
493                     queryRevision += "\n";
494 
495                     // Delete old entries (added)
496                     queryRevision += String.format(
497                             "DELETE { GRAPH <%s> { %n"
498                                     + "	<%s> rpo:added ?blank . %n"
499                                     + "	?blank rdf:subject ?s . %n"
500                                     + "	?blank rdf:predicate ?p . %n"
501                                     + "	?blank rdf:object ?o . %n"
502                                     + "	?blank rmo:references ?revision . %n"
503                                     + "} } %n"
504                                     + "WHERE { "

661     /**
662      * Create the difference triple model which contains all differing triples.
663      *
664      * @param graphName the graph name
665      * @param graphNameDifferenceTripleModel the graph name of the difference triple model
666      * @param graphNameRevisionProgressA the graph name of the revision progress of branch A
667      * @param uriA the URI of the revision progress of branch A
668      * @param graphNameRevisionProgressB the graph name of the revision progress of branch B
669      * @param uriB the URI of the revision progress of branch B
670      * @param uriSDD the URI of the SDD to use
671      */
672     private void createDifferenceTripleModel(String graphName, String graphNameDifferenceTripleModel, String graphNameRevisionProgressA, String uriA, String graphNameRevisionProgressB, String uriB, String uriSDD){
673 
674         logger.info("Create the difference triple model");
675         TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format("DROP SILENT GRAPH <%s>", graphNameDifferenceTripleModel));
676         TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format("CREATE GRAPH  <%s>", graphNameDifferenceTripleModel));
677 
678         // Templates for revision A and B
679         String sparqlTemplateRevisionA = String.format(
680                 "	GRAPH <%s> { %n"
681                         + "		<%s> <%s> ?blankA . %n"
682                         + "			?blankA rdf:subject ?s . %n"
683                         + "			?blankA rdf:predicate ?p . %n"
684                         + "			?blankA rdf:object ?o . %n"
685                         + "			?blankA rmo:references ?revisionA . %n"
686                         + "	} %n", graphNameRevisionProgressA, uriA, "%s");
687         String sparqlTemplateRevisionB = String.format(
688                 "	GRAPH <%s> { %n"
689                         + "		<%s> <%s> ?blankB . %n"
690                         + "			?blankB rdf:subject ?s . %n"
691                         + "			?blankB rdf:predicate ?p . %n"
692                         + "			?blankB rdf:object ?o . %n"
693                         + "			?blankB rmo:references ?revisionB . %n"
694                         + "	} %n", graphNameRevisionProgressB, uriB, "%s");
695 
696         String sparqlTemplateNotExistsRevisionA = String.format(
697                 "FILTER NOT EXISTS { %n"
698                         + "	GRAPH <%s> { %n"
699                         + "		<%s> ?everything ?blankA . %n"
700                         + "			?blankA rdf:subject ?s . %n"
701                         + "			?blankA rdf:predicate ?p . %n"
702                         + "			?blankA rdf:object ?o . %n"
703                         + "			?blankA rmo:references ?revisionA . %n"
704                         + "	} %n"
705                         + "}", graphNameRevisionProgressA, uriA);
706 
707         String sparqlTemplateNotExistsRevisionB = String.format(
708                 "FILTER NOT EXISTS { %n"
709                         + "	GRAPH <%s> { %n"
710                         + "		<%s> ?everything ?blankB . %n"
711                         + "			?blankB rdf:subject ?s . %n"
712                         + "			?blankB rdf:predicate ?p . %n"
713                         + "			?blankB rdf:object ?o . %n"
714                         + "			?blankB rmo:references ?revisionB . %n"
715                         + "	} %n"
716                         + "}", graphNameRevisionProgressB, uriB);
717 
718         // Get all structural definitions which are generating differences
719         String queryDifferingSD = String.format(
720                 "PREFIX sddo: <http://eatld.et.tu-dresden.de/sddo#> %n"
721                         + "PREFIX sdd:  <http://eatld.et.tu-dresden.de/sdd#> %n"
722                         + "PREFIX xsd:  <http://www.w3.org/2001/XMLSchema#> %n"
723                         + "SELECT ?combinationURI ?tripleStateA ?tripleStateB ?conflict ?automaticResolutionState %n"
724                         + "WHERE { GRAPH <%s> { %n"
725                         + "	<%s> a sddo:StructuralDefinitionGroup ;"
726                         + "		sddo:hasStructuralDefinition ?combinationURI ."
727                         + "	?combinationURI a sddo:StructuralDefinition ; %n"
728                         + "		sddo:hasTripleStateA ?tripleStateA ; %n"
729                         + "		sddo:hasTripleStateB ?tripleStateB ; %n"
730                         + "		sddo:isConflicting ?conflict ; %n"
731                         + "		sddo:automaticResolutionState ?automaticResolutionState . %n"
732                         + "} } %n", Config.sdd_graph, uriSDD);
733 
734         // Iterate over all differing combination URIs
735         ResultSet resultSetDifferences = TripleStoreInterfaceSingleton.get().executeSelectQuery(queryDifferingSD);
736         while (resultSetDifferences.hasNext()) {
737             QuerySolution qs = resultSetDifferences.next();
738 
739             String currentDifferenceCombinationURI = qs.getResource("?combinationURI").toString();
740             String currentTripleStateA = qs.getResource("?tripleStateA").toString();
741             String currentTripleStateB = qs.getResource("?tripleStateB").toString();
742             // Will return an integer value because virtuoso stores boolean internal as integer
743             String currentConflictState = qs.getLiteral("?conflict").toString();
744             // TDB returns boolean value without "" -> add it to use it in the next query correctly
745             if (currentConflictState.equals("true^^http://www.w3.org/2001/XMLSchema#boolean")) {
746                 currentConflictState = "\"true\"^^<http://www.w3.org/2001/XMLSchema#boolean>";
747             } else {
748                 currentConflictState = "\"false\"^^<http://www.w3.org/2001/XMLSchema#boolean>";
749             }
750             String currentAutomaticResolutionState = qs.getResource("?automaticResolutionState").toString();
751 
752             String querySelectPart = "SELECT ?s ?p ?o %s %s %n";
753             String sparqlQueryRevisionA = null;
754             String sparqlQueryRevisionB = null;
755 
756             // A
757             if (currentTripleStateA.equals(SDDTripleStateEnum.ADDED.getSddRepresentation())) {
758                 // In revision A the triple was added
759                 querySelectPart = String.format(querySelectPart, "?revisionA", "%s");
760                 sparqlQueryRevisionA = String.format(sparqlTemplateRevisionA, SDDTripleStateEnum.ADDED.getRpoRepresentation());
761             } else if (currentTripleStateA.equals(SDDTripleStateEnum.DELETED.getSddRepresentation())) {
762                 // In revision A the triple was deleted
763                 querySelectPart = String.format(querySelectPart, "?revisionA", "%s");
764                 sparqlQueryRevisionA = String.format(sparqlTemplateRevisionA, SDDTripleStateEnum.DELETED.getRpoRepresentation());
765             } else if (currentTripleStateA.equals(SDDTripleStateEnum.ORIGINAL.getSddRepresentation())) {
766                 // In revision A the triple is original
767                 querySelectPart = String.format(querySelectPart, "?revisionA", "%s");
768                 sparqlQueryRevisionA = String.format(sparqlTemplateRevisionA, SDDTripleStateEnum.ORIGINAL.getRpoRepresentation());
769             } else if (currentTripleStateA.equals(SDDTripleStateEnum.NOTINCLUDED.getSddRepresentation())) {
770                 // In revision A the triple is not included
771                 querySelectPart = String.format(querySelectPart, "", "%s");
772                 sparqlQueryRevisionA = sparqlTemplateNotExistsRevisionA;
773             }
774 
775             // B
776             if (currentTripleStateB.equals(SDDTripleStateEnum.ADDED.getSddRepresentation())) {
777                 // In revision B the triple was added
778                 querySelectPart = String.format(querySelectPart, "?revisionB");
779                 sparqlQueryRevisionB = String.format(sparqlTemplateRevisionB, SDDTripleStateEnum.ADDED.getRpoRepresentation());
780             } else if (currentTripleStateB.equals(SDDTripleStateEnum.DELETED.getSddRepresentation())) {
781                 // In revision B the triple was deleted
782                 querySelectPart = String.format(querySelectPart, "?revisionB");
783                 sparqlQueryRevisionB = String.format(sparqlTemplateRevisionB, SDDTripleStateEnum.DELETED.getRpoRepresentation());
784             } else if (currentTripleStateB.equals(SDDTripleStateEnum.ORIGINAL.getSddRepresentation())) {
785                 // In revision B the triple is original
786                 querySelectPart = String.format(querySelectPart, "?revisionB");
787                 sparqlQueryRevisionB = String.format(sparqlTemplateRevisionB, SDDTripleStateEnum.ORIGINAL.getRpoRepresentation());
788             } else if (currentTripleStateB.equals(SDDTripleStateEnum.NOTINCLUDED.getSddRepresentation())) {
789                 // In revision B the triple is not included
790                 querySelectPart = String.format(querySelectPart, "");
791                 sparqlQueryRevisionB = sparqlTemplateNotExistsRevisionB;
792             }
793 
794             // Concatenated SPARQL query
795             String query = String.format(
796                     Config.prefixes
797                             + "%s"
798                             + "WHERE { %n"
799                             + "%s"
800                             + "%s"
801                             + "} %n", querySelectPart, sparqlQueryRevisionA, sparqlQueryRevisionB);
802 
803             // Iterate over all triples
804             ResultSet resultSetTriples = TripleStoreInterfaceSingleton.get().executeSelectQuery(query);
805             while (resultSetTriples.hasNext()) {
806                 QuerySolution qsQuery = resultSetTriples.next();
807 
808                 String subject = qsQuery.getResource("?s").toString();
809                 String predicate = qsQuery.getResource("?p").toString();
810 
811                 // Differ between literal and resource
812                 String object = "";
813                 if (qsQuery.get("?o").isLiteral()) {
814                     object = "\"" + qsQuery.getLiteral("?o").toString() + "\"";
815                 } else {
816                     object = "<" + qsQuery.getResource("?o").toString() + ">";
817                 }
818 
819                 // Create the references A and B part of the query
820                 String referencesAB = ". %n";
821                 if (!currentTripleStateA.equals(SDDTripleStateEnum.NOTINCLUDED.getSddRepresentation()) && !currentTripleStateB.equals(SDDTripleStateEnum.NOTINCLUDED.getSddRepresentation())) {
822                     referencesAB = String.format(
823                             "			rpo:referencesA <%s> ; %n"
824                                     + "			rpo:referencesB <%s> %n", qsQuery.getResource("?revisionA").toString(),
825                             qsQuery.getResource("?revisionB").toString());
826                 } else if (currentTripleStateA.equals(SDDTripleStateEnum.NOTINCLUDED.getSddRepresentation()) && !currentTripleStateB.equals(SDDTripleStateEnum.NOTINCLUDED.getSddRepresentation())) {
827                     referencesAB = String.format(
828                             "			rpo:referencesB <%s> %n", qsQuery.getResource("?revisionB").toString());
829                 } else if (!currentTripleStateA.equals(SDDTripleStateEnum.NOTINCLUDED.getSddRepresentation()) && currentTripleStateB.equals(SDDTripleStateEnum.NOTINCLUDED.getSddRepresentation())) {
830                     referencesAB = String.format(
831                             "			rpo:referencesA <%s> %n", qsQuery.getResource("?revisionA").toString());
832                 }
833 
834                 String queryTriple = Config.prefixes + String.format(
835                         "INSERT DATA { GRAPH <%s> {%n"
836                                 + "	<%s> a rpo:DifferenceGroup ; %n"
837                                 + "	sddo:hasTripleStateA <%s> ; %n"
838                                 + "	sddo:hasTripleStateB <%s> ; %n"
839                                 + "	sddo:isConflicting %s ; %n"
840                                 + "	sddo:automaticResolutionState <%s> ; %n"
841                                 + "	rpo:hasDifference [ %n"
842                                 + "		a rpo:Difference ; %n"
843                                 + "			rpo:hasTriple [ %n"
844                                 + "				rdf:subject <%s> ; %n"
845                                 + "				rdf:predicate <%s> ; %n"
846                                 + "				rdf:object %s %n"
847                                 + "			] ; %n"
848                                 + "%s"
849                                 + "	] . %n"
850                                 + "} }", graphNameDifferenceTripleModel,
851                         currentDifferenceCombinationURI,
852                         currentTripleStateA,
853                         currentTripleStateB,
854                         currentConflictState,
855                         currentAutomaticResolutionState,
856                         subject,
857                         predicate,
858                         object,
859                         referencesAB);
860 
861                 TripleStoreInterfaceSingleton.get().executeUpdateQuery(queryTriple);
862             }
863         }
864     }
