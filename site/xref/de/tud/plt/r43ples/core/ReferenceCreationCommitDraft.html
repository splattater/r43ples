15  /**
16   * Collection of information for creating a new reference creation commit.
17   *
18   * @author Stephan Hensel
19   */
20  public class ReferenceCreationCommitDraft extends CommitDraft {
21  
22      /** The logger. **/
23      private Logger logger = Logger.getLogger(ReferenceCreationCommitDraft.class);
24  
25      /** The pattern modifier. **/
26      private final int patternModifier = Pattern.DOTALL + Pattern.MULTILINE + Pattern.CASE_INSENSITIVE;
27  
28      /** The graph name. **/
29      private String graphName;
30      /** The reference name. **/
31      private String referenceName;
32      /** The revision identifier. **/
33      private String revisionIdentifier;
34      /** States if the created reference is a branch or a tag. (branch => true; tag => false) **/
35      private boolean isBranch;
36  
37      /** States if this commit draft was created by a request or add and delete sets. (true => request, false => add/delete sets) **/
38      private boolean isCreatedWithRequest;
39  
40  
41      /**
42       * The constructor.
43       *
44       * @param request the request received by R43ples
45       * @throws InternalErrorException
46       */
47      public ReferenceCreationCommitDraft(R43plesRequest request) throws InternalErrorException {
48          super(request);
49          this.extractRequestInformation();
50          this.isCreatedWithRequest = true;
51      }
52  
53      /**
54       * The constructor.
55       * Creates an reference creation commit draft by using the corresponding meta information.
56       *
57       * @param graphName the graph name
58       * @param referenceName the reference name
59       * @param revisionIdentifier the revision identifier (the corresponding revision will be the current base for the reference)
60       * @param user the user
61       * @param message the message
62       * @param isBranch states if the created reference is a branch or a tag. (branch => true; tag => false)
63       * @throws InternalErrorException
64       */
65      protected ReferenceCreationCommitDraft(String graphName, String referenceName, String revisionIdentifier, String user, String message, boolean isBranch) throws InternalErrorException {
66          super(null);
67          this.setUser(user);
68          this.setMessage(message);
69          this.graphName = graphName;
70          this.referenceName = referenceName;
71          this.revisionIdentifier = revisionIdentifier;
72          this.isBranch = isBranch;
73          this.isCreatedWithRequest = false;
74      }
75  
76      /**
77       * Extracts the request information and stores it to local variables.
78       *
79       * @throws InternalErrorException
80       */
81      private void extractRequestInformation() throws InternalErrorException {
82          final Pattern patternBranchOrTagQuery = Pattern.compile("(?<action>TAG|BRANCH)\\s*GRAPH\\s*<(?<graph>[^>]*)>\\s*REVISION\\s*\"(?<revision>[^\"]*)\"\\s*TO\\s*\"(?<name>[^\"]*)\"",
83                  patternModifier);
84          Matcher m = patternBranchOrTagQuery.matcher(getRequest().query_sparql);
85  
86          boolean foundEntry = false;
87  
88          while (m.find()) {
89              foundEntry = true;
90              String action = m.group("action");
91              this.graphName = m.group("graph");
92              this.revisionIdentifier = m.group("revision").toLowerCase();
93              this.referenceName = m.group("name").toLowerCase();
94              if (action.equals("TAG")) {
95                  this.isBranch = false;
96              } else if (action.equals("BRANCH")) {
97                  this.isBranch = true;
98              } else {
99                  throw new QueryErrorException("Error in query: " + getRequest().query_sparql);
100             }
101         }
102         if (!foundEntry) {
103             throw new QueryErrorException("Error in query: " + getRequest().query_sparql);
104         }
105     }

107     /**
108      * Creates the commit draft as a new commit in the triple store.
109      *
110      * @return the created reference commit
111      */
112     protected ReferenceCommit createCommitInTripleStore() throws InternalErrorException {
113         RevisionGraph revisionGraph = new RevisionGraph(graphName);
114         String referenceURI;
115         String commitURI;
116         if (isBranch) {
117             referenceURI = getRevisionManagement().getNewBranchURI(revisionGraph, referenceName);
118             commitURI = getRevisionManagement().getNewBranchCommitURI(revisionGraph, referenceName);
119         } else {
120             referenceURI = getRevisionManagement().getNewTagURI(revisionGraph, referenceName);
121             commitURI = getRevisionManagement().getNewTagCommitURI(revisionGraph, referenceName);
122         }
123 
124         addMetaInformation(referenceURI, commitURI);
125 
126         Revision usedRevision = new Revision(revisionGraph, revisionIdentifier, true);
127         Reference generatedReference;
128         if (isBranch) {
129             generatedReference = new Branch(revisionGraph, referenceName, referenceURI);
130         } else {
131             generatedReference = new Tag(revisionGraph, referenceName, referenceURI);
132         }
133 
134         return new ReferenceCommit(revisionGraph, commitURI, getUser(), getTimeStamp(), getMessage(), usedRevision, generatedReference);
135     }

137     /**
138      * Creates a new full graph for the new reference and and adds the necessary meta data.
139      *
140      * @param referenceURI the reference URI
141      * @param commitURI the commit URI
142      * @throws InternalErrorException
143      */
144     private void addMetaInformation(String referenceURI, String commitURI) throws InternalErrorException {
145         if (isBranch) {
146             logger.info("Create new branch '" + referenceName + "' for graph " + graphName);
147         } else {
148             logger.info("Create new tag '" + referenceName + "' for graph " + graphName);
149         }
150 
151         RevisionGraph graph = new RevisionGraph(graphName);
152         String revisionGraph = graph.getRevisionGraphUri();
153         // Check branch existence
154         if (graph.hasReference(referenceName)) {
155             // Reference name is already in use
156             logger.error("The reference name '" + referenceName + "' is for the graph '" + graphName
157                     + "' already in use.");
158             throw new IdentifierAlreadyExistsException("The reference name '" + referenceName
159                     + "' is for the graph '" + graphName + "' already in use.");
160         } else {
161             // General variables
162             String referenceTypeUri = isBranch ? "rmo:Branch" : "rmo:Tag";
163             String revisionUri = graph.getRevisionUri(revisionIdentifier);
164             String personUri = RevisionManagementOriginal.getUserURI(getUser());
165 
166             // Create a new commit (activity)
167             String queryContent = String.format(""
168                             + "<%s> a %sCommit, rmo:Commit; "
169                             + "	prov:wasAssociatedWith <%s> ;"
170                             + "	prov:generated <%s> ;"
171                             + " prov:used <%s> ;"
172                             + "	dc-terms:title \"%s\" ;"
173                             + "	prov:atTime \"%s\" .%n",
174                     commitURI, referenceTypeUri, personUri, referenceURI, revisionUri, getMessage(), getTimeStamp());
175 
176             // Create new reference (branch/tag)
177             queryContent += String.format(""
178                             + "<%s> a %s, rmo:Reference; "
179                             + " rmo:fullGraph <%s>; "
180                             + "	prov:wasDerivedFrom <%s>; "
181                             + "	rmo:references <%s>; "
182                             + "	rdfs:label \"%s\". ",
183                     referenceURI, referenceTypeUri, referenceURI, revisionUri, revisionUri, referenceName);
184 
185             // Update full graph of branch
186             RevisionManagementOriginal.generateFullGraphOfRevision(graphName, revisionIdentifier, referenceURI);
187 
188             // Execute queries
189             String query = Config.prefixes
190                     + String.format("INSERT DATA { GRAPH <%s> { %s } } ;", revisionGraph, queryContent);
191             getTripleStoreInterface().executeUpdateQuery(query);
192         }
193     }
